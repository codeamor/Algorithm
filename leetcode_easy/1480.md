# Running Sum of 1d Array

```js
var runningSum = function(nums) {
    let sum = 0
    let sumArr = []
    
    for (let i = 0; i < nums.length; i++) {
        sum += nums[i]
        sumArr.push(sum)
    }
    return sumArr
};
```

```js
var runningSum = function(nums) {
   for (let i = 1; i < nums.length; i++) {
        nums[i] += nums[i - 1]
    }
    return nums;
};
```
